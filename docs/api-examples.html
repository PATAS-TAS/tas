🛡️ TAS - Transmodal Anti-Spam API - Examples

1. Single Text Classification

POST /classify

curl -X POST https://tas.fly.dev/classify \
  -H "Content-Type: application/json" \
  -d '{
    "text": "Sell iPhone 12, cheap price, call me +79001234567",
    "lang": "en"
  }'

Response:

{
  "spam_score": 0.85,
  "confidence": 0.87,
  "labels": ["spam", "scam"],
  "reasons": ["Commercial trade offer", "Contains phone number"],
  "layers_used": ["rules"],
  "version": "1.0.0"
}

5. JavaScript Example

async function checkSpam(text) {
  const response = await fetch('https://tas.fly.dev/classify', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      text: text,
      lang: 'en'
    })
  });
  
  const result = await response.json();
  return result;
}

// Usage
const result = await checkSpam("Sell iPhone 12, cheap!");
console.log(result.spam_score); // 0.85
console.log(result.labels); // ["spam"]

6. Python Example

import requests

def check_spam(text):
    response = requests.post(
        'https://tas.fly.dev/classify',
        json={'text': text, 'lang': 'en'}
    )
    return response.json()

# Usage
result = check_spam("Sell iPhone 12, cheap!")
print(f"Spam score: {result['spam_score']}")
print(f"Labels: {result['labels']}")

7. Rate Limiting

For production use, implement rate limiting on your side:

// Simple rate limiter
class RateLimiter {
  constructor(maxRequests, windowMs) {
    this.requests = [];
    this.maxRequests = maxRequests;
    this.windowMs = windowMs;
  }
  
  canMakeRequest() {
    const now = Date.now();
    this.requests = this.requests.filter(
      time => now - time < this.windowMs
    );
    
    if (this.requests.length < this.maxRequests) {
      this.requests.push(now);
      return true;
    }
    return false;
  }
}
